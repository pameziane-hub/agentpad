## Kern-Schnittstellen (verbindlich, damit Tasks zusammenpassen)

// Curves.swift
public enum Curves {

    public static func shape(x: Float, y: Float, deadzone: Float, expo: Float) -> SIMD2<Float>

// KeyCombo.swift
public struct KeyCombo: Equatable { public let keyCode: UInt16; public let flags: KeyFlags }
public struct KeyFlags: OptionSet { command, shift, control, option }   // mapped auf CGEventFlags in OutputService
public enum KeyComboParser {
    public static func parse(_ raw: String) -> KeyCombo?            // "cmd+`", "esc", "ctrl"
    public static func parseSequence(_ raw: String) -> [KeyCombo]?  // space-getrennt: "ctrl ctrl"
    public static func isModifierOnly(_ combo: KeyCombo) -> Bool    // ctrl/shift/cmd/opt-Keycodes
}
// Keycode-Tabelle (kVK, layout-positionsbasiert): return 36, tab 48, space 49, esc 53,
// ` 50, ctrl 59, left 123, right 124, down 125, up 126, Ziffern 1-9,0: 18,19,20,21,23,22,26,28,25,29

// Config.swift
public struct Config: Codable, Equatable {
    public var pointer: PointerConfig   // deadzone 0.12, expo 0.6, maxSpeed 1400 px/s
    public var scroll: ScrollConfig     // deadzone 0.15, speed 600 px/s
    public var buttons: [String: ButtonAction]
    public static let `default`: Config
}

    case key(String), url(String), leftClick, rightClick, pause

public enum ConfigLoader {

**Default-Mapping** (Button-IDs = Keys im `buttons`-Dict):
`a`‚Üíkey return ¬∑ `b`‚Üíkey esc ¬∑ `x`‚Üíkey tab ¬∑ `y`‚Üíkey shift+tab ¬∑ `dpadUp/Down/Left/Right`‚ÜíPfeile ¬∑ `leftShoulder`‚Üíkey cmd+\` ¬∑ `rightShoulder`‚Üíkey "ctrl ctrl" (macOS-Diktat) ¬∑ `leftTrigger`‚ÜírightClick ¬∑ `rightTrigger`‚ÜíleftClick ¬∑ `menu`‚Üípause

**App-Schicht-Verhalten:**
- `ControllerService`: `GCControllerDidConnect/Disconnect`-Notifications (main queue), `pressedChangedHandler` je Button ‚Üí `onButton(id, pressed)`, `valueChangedHandler` der Thumbsticks ‚Üí gepufferte `leftStick`/`rightStick`-SIMD2-Werte. Disconnect nullt Sticks.
- `OutputService`: CGEventSource `.hidSystemState`. Maus: aktuelle Position via `CGEvent(source:nil).location`, Ziel an aktive Displays clampen (`CGGetActiveDisplayList`/`CGDisplayBounds`), Typ `.leftMouseDragged` wenn Linksklick gehalten (‚Üí Drag). Scroll: `scrollWheelEvent2Source`, Pixel-Units, wheel1=vertikal/wheel2=horizontal. Keys: keyDown+keyUp mit Flags; **Modifier-only-Combos** (ctrl f√ºr Diktat) als `.flagsChanged`-Events, Sequenz-Glieder mit 60 ms Abstand (Double-Tap-Erkennung). URL: `NSWorkspace.shared.open`.
- `Engine`: Timer 120 Hz auf Main-RunLoop (`.common`). - `MenuBarController`: NSStatusItem, SF-Symbol `gamecontroller.fill` (aktiv) / `gamecontroller` (getrennt) / `pause.circle` (Pause) / `exclamationmark.triangle` (keine AX-Permission). Men√º: Statuszeile (Controller-Name), Pause ‚å•-toggle, ‚ÄûOpen Config", Quit.
